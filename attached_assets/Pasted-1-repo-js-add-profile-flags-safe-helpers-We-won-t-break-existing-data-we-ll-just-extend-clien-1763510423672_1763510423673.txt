We won’t break existing data; we’ll just extend client objects.

In repo.js, find your createClient function. It’ll look something like:

Change it to include profileComplete and onboardingStep:

async function createClient(data) {
  const id = data.id || ('cl_' + nid());
  const client = {
    id,
    businessId: data.businessId,
    name: data.name || '',
    email: data.email || '',
    phone: data.phone || '',
    address: data.address || '',
    notes: data.notes || '',
    profileComplete: false,
    onboardingStep: data.onboardingStep || 1,
    createdAt: isoNow(),
    updatedAt: isoNow()
  };
  db.clients[id] = client;
  return client;
}

Now add two helper functions near your other client functions (e.g. near getClient / updateClient):

async function updateClient(id, patch) {
  const existing = db.clients[id];
  if (!existing) return null;
  const updated = {
    ...existing,
    ...patch,
    updatedAt: isoNow()
  };
  db.clients[id] = updated;
  return updated;
}

async function markClientProfileComplete(id) {
  return updateClient(id, {
    profileComplete: true,
    onboardingStep: 999
  });
}

If you already have an updateClient, just add the markClientProfileComplete and ensure it preserves profileComplete.

Export them at the bottom where other repo methods are exported:

export const repo = {

  createClient,
  updateClient,
  markClientProfileComplete,
  // ...
};

import React, { useEffect, useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { repo } from '../../repo.js';

const STEPS = [
  'Your details',
  'Address and access',
  'Emergency contact',
  'Your dogs',
  'Dog behaviour and vet',
  'Review and complete'
];

export function ClientOnboarding() {
  const navigate = useNavigate();
  const [client, setClient] = useState(null);
  const [dogs, setDogs] = useState([]);
  const [step, setStep] = useState(1);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);

  useEffect(() => {
    (async () => {
      const raw = localStorage.getItem('pt_client');
      if (!raw) {
        navigate('/client/login');
        return;
      }
      const { clientId, businessId } = JSON.parse(raw);

      const [c, allDogs] = await Promise.all([
        repo.getClient(clientId),
        repo.listDogsByBusiness(businessId)
      ]);

      if (!c) {
        localStorage.removeItem('pt_client');
        navigate('/client/login');
        return;
      }

      setClient(c);
      setDogs(allDogs.filter(d => d.clientId === c.id));
      setStep(c.onboardingStep || 1);
      setLoading(false);
    })();
  }, [navigate]);

  async function handleSaveStep(patch, nextStep) {
    if (!client) return;
    setSaving(true);
    const updated = await repo.updateClient(client.id, {
      ...patch,
      onboardingStep: nextStep
    });
    setClient(updated);
    setStep(nextStep);
    setSaving(false);
  }

  async function handleComplete() {
    if (!client) return;
    setSaving(true);
    const updated = await repo.markClientProfileComplete(client.id);
    setClient(updated);
    setSaving(false);
    navigate('/client/dashboard');
  }

  if (loading) {
    return (
      <div className="max-w-xl mx-auto py-8">
        <p className="text-sm text-slate-600">Loading your details…</p>
      </div>
    );
  }

  if (!client) return null;

  return (
    <div className="max-w-2xl mx-auto py-8 space-y-6">
      <div>
        <h1 className="text-xl font-semibold">Set up your account</h1>
        <p className="text-sm text-slate-600">
          A few steps so your walker has everything they need to look after your dog safely.
        </p>
      </div>

      <Progress steps={STEPS} current={step} />

      <div className="card space-y-4">
        {step === 1 && (
          <StepDetails client={client} onSave={handleSaveStep} saving={saving} />
        )}
        {step === 2 && (
          <StepAddress client={client} onSave={handleSaveStep} saving={saving} />
        )}
        {step === 3 && (
          <StepEmergency client={client} onSave={handleSaveStep} saving={saving} />
        )}
        {step === 4 && (
          <StepDogs client={client} dogs={dogs} setDogs={setDogs} saving={saving} onSave={handleSaveStep} />
        )}
        {step === 5 && (
          <StepDogBehaviour client={client} dogs={dogs} saving={saving} onSave={handleSaveStep} />
        )}
        {step === 6 && (
          <StepReview client={client} dogs={dogs} onComplete={handleComplete} saving={saving} />
        )}
      </div>
    </div>
  );
}

function Progress({ steps, current }) {
  return (
    <ol className="grid grid-cols-3 gap-2 text-xs text-slate-600 md:grid-cols-6">
      {steps.map((label, index) => {
        const stepNumber = index + 1;
        const active = stepNumber === current;
        const done = stepNumber < current;
        return (
          <li key={label} className="flex items-center gap-2">
            <span
              className={
                'inline-flex h-5 w-5 items-center justify-center rounded-full border text-[11px] ' +
                (done
                  ? 'bg-teal-600 border-teal-600 text-white'
                  : active
                  ? 'bg-teal-100 border-teal-600 text-teal-700'
                  : 'bg-white border-slate-300 text-slate-500')
              }
            >
              {stepNumber}
            </span>
            <span className={active ? 'font-medium' : ''}>{label}</span>
          </li>
        );
      })}
    </ol>
  );
}

3. ClientGuard – block booking until onboarding complete

Create src/components/ClientGuard.jsx:

import React, { useEffect, useState } from 'react';
import { Navigate, useLocation } from 'react-router-dom';
import { repo } from '../../repo.js';

export function ClientGuard({ children }) {
  const location = useLocation();
  const [checked, setChecked] = useState(false);
  const [allowed, setAllowed] = useState(false);

  useEffect(() => {
    (async () => {
      const raw = localStorage.getItem('pt_client');
      if (!raw) {
        setAllowed(false);
        setChecked(true);
        return;
      }
      const { clientId } = JSON.parse(raw);
      const client = await repo.getClient(clientId);
      if (!client) {
        setAllowed(false);
        setChecked(true);
        return;
      }

      // Allow onboarding route even if incomplete.
      const isOnboarding = location.pathname.startsWith('/client/onboarding');

      if (client.profileComplete || isOnboarding) {
        setAllowed(true);
      } else {
        setAllowed(false);
      }
      setChecked(true);
    })();
  }, [location.pathname]);

  if (!checked) {
    return <p className="text-sm text-slate-600 px-4 py-6">Checking your account…</p>;
  }

  if (!allowed && !location.pathname.startsWith('/client/onboarding')) {
    return <Navigate to="/client/onboarding" replace />;
  }

  return children;
}

import React, { useEffect, useMemo, useState } from 'react';
import { repo } from '../../repo.js';

export function AdminDashboard() {
  const [loading, setLoading] = useState(true);
  const [jobs, setJobs] = useState([]);
  const [clients, setClients] = useState([]);
  const [staff, setStaff] = useState([]);

  useEffect(() => {
    (async () => {
      // You likely have auth.user.businessId already; if not, adjust.
      const raw = localStorage.getItem('pt_user');
      if (!raw) {
        setLoading(false);
        return;
      }
      const u = JSON.parse(raw);
      if (!u.businessId) {
        setLoading(false);
        return;
      }

      const [j, c, s] = await Promise.all([
        repo.listJobsByBusiness(u.businessId),
        repo.listClientsByBusiness(u.businessId),
        repo.listStaffByBusiness(u.businessId)
      ]);

      setJobs(j);
      setClients(c);
      setStaff(s);
      setLoading(false);
    })();
  }, []);

  const metrics = useMemo(() => {
    const now = new Date();
    const startOfWeek = new Date(now);
    startOfWeek.setDate(startOfWeek.getDate() - startOfWeek.getDay() + 1);
    startOfWeek.setHours(0, 0, 0, 0);

    const endOfWeek = new Date(startOfWeek);
    endOfWeek.setDate(endOfWeek.getDate() + 7);

    let jobsThisWeek = 0;
    let jobsNextWeek = 0;

    for (const j of jobs) {
      if (!j.start) continue;
      const d = new Date(j.start);
      if (d >= startOfWeek && d < endOfWeek) jobsThisWeek++;
      if (d >= endOfWeek && d < new Date(endOfWeek.getTime() + 7 * 86400000)) {
        jobsNextWeek++;
      }
    }

    return {
      totalClients: clients.length,
      activeClients: clients.filter(c => c.profileComplete).length,
      staffCount: staff.length,
      jobsThisWeek,
      jobsNextWeek
    };
  }, [jobs, clients, staff]);

  if (loading) {
    return <p className="text-sm text-slate-600">Loading dashboard…</p>;
  }

  return (
    <div className="space-y-6">
      <header>
        <h1 className="text-xl font-semibold">Overview</h1>
        <p className="text-sm text-slate-600">
          A high-level view of clients, staff and bookings.
        </p>
      </header>

      <section className="grid gap-4 md:grid-cols-3">
        <StatCard
          label="Total clients"
          value={metrics.totalClients}
        />
        <StatCard
          label="Active clients"
          value={metrics.activeClients}
        />
        <StatCard
          label="Staff"
          value={metrics.staffCount}
        />
      </section>

      <section className="grid gap-4 md:grid-cols-2">
        <StatCard
          label="Jobs this week"
          value={metrics.jobsThisWeek}
        />
        <StatCard
          label="Jobs next week"
          value={metrics.jobsNextWeek}
        />
      </section>

      {/* Placeholder for charts; we will fill with Chart.js later */}
      <section className="grid gap-4 md:grid-cols-2">
        <div className="card h-48 flex items-center justify-center text-sm text-slate-500">
          Jobs over time (chart)
        </div>
        <div className="card h-48 flex items-center justify-center text-sm text-slate-500">
          Service breakdown (chart)
        </div>
      </section>
    </div>
  );
}

function StatCard({ label, value }) {
  return (
    <div className="card">
      <div className="text-xs text-slate-500">{label}</div>
      <div className="mt-2 text-2xl font-semibold text-slate-900">
        {value}
      </div>
    </div>
  );
}
